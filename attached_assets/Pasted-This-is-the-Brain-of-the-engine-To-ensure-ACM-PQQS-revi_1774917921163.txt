package scanner

import (
	"fmt"
	"go/ast"
	"go/parser"
	"go/token"
	"os"
	"path/filepath"
)

// Finding represents a discovered cryptographic primitive
type Finding struct {
	File      string
	Line      int
	Algorithm string
	Risk      string // Quantum-Vulnerable or PQC-Ready
}

// PerformScan recursively traverses the target path for cryptographic nodes
func PerformScan(rootPath string) ([]Finding, error) {
	var findings []Finding
	fset := token.NewFileSet()

	err := filepath.Walk(rootPath, func(path string, info os.FileInfo, err error) error {
		if err != nil {
			return err
		}

		// Only scan .go files, skipping vendor and hidden directories
		if info.IsDir() || filepath.Ext(path) != ".go" {
			return nil
		}

		// Parse the file into an Abstract Syntax Tree (AST)
		node, err := parser.ParseFile(fset, path, nil, parser.ParseComments)
		if err != nil {
			return nil // Skip files that don't parse
		}

		// Walk the AST looking for specific crypto patterns
		ast.Inspect(node, func(n ast.Node) bool {
			call, ok := n.(*ast.CallExpr)
			if !ok {
				return true
			}

			// Look for selectors like rsa.GenerateKey or ecdsa.GenerateKey
			if sel, ok := call.Fun.(*ast.SelectorExpr); ok {
				if x, ok := sel.X.(*ast.Ident); ok {
					
					// FLAG: Legacy RSA (NIST FIPS 186-4)
					if x.Name == "rsa" && sel.Sel.Name == "GenerateKey" {
						findings = append(findings, Finding{
							File:      path,
							Line:      fset.Position(n.Pos()).Line,
							Algorithm: "RSA-Legacy",
							Risk:      "Quantum-Vulnerable (HNDL Risk)",
						})
					}

					// FLAG: Legacy ECDSA (NIST SP 800-186)
					if x.Name == "ecdsa" && sel.Sel.Name == "GenerateKey" {
						findings = append(findings, Finding{
							File:      path,
							Line:      fset.Position(n.Pos()).Line,
							Algorithm: "ECDSA-Legacy",
							Risk:      "Quantum-Vulnerable (HNDL Risk)",
						})
					}
				}
			}
			return true
		})

		return nil
	})

	return findings, err
}
